14   class GameActivity : AppCompatActivity() {
15   
16       lateinit var textScore : TextView
17       lateinit var textLife : TextView
18       lateinit var textTime : TextView
19   
20       lateinit var textQuestion : TextView
21       lateinit var editTextAnswer : EditText
22   
23       lateinit var buttonOk : Button
24       lateinit var buttonNext : Button
25   
26       var correctanswer = 0
27       var userscore = 0
28       var userlife = 3
29   
30       lateinit var timer : CountDownTimer
31       private val startTimerInmillis : Long = 60000
32       var timeLeftInmillis : Long = startTimerInmillis
33   
34   
35       override fun onCreate(savedInstanceState: Bundle?) {
36           super.onCreate(savedInstanceState)
37           setContentView(R.layout.activity_game)
38   
39           supportActionBar!!.title = "Addition"
40   
41           textScore = findViewById(R.id.textViewScore)
42           textLife = findViewById(R.id.textViewLife)
43           textTime = findViewById(R.id.textViewTime)
44           textQuestion = findViewById(R.id.textViewQuestion)
45           editTextAnswer = findViewById(R.id.editTextAnswer)
46           buttonOk = findViewById(R.id.buttonOk)
47           buttonNext = findViewById(R.id.buttonNext)
48   
49           gameContinue()
50   
51           buttonOk.setOnClickListener {
52               val input = editTextAnswer.text.toString()
53               if (input == "")
54               {
55                   Toast.makeText(applicationContext,"Please write an answer or click the next button",Toast.LENGTH_LONG).show()
56               }
57               else
58               {
59                   pauseTimer()
60                   val userAnswer = input.toInt()
61                   if(userAnswer == correctanswer)
62                   {
63                       userscore += 10
64                       textQuestion.text = "Congratulations,your answer is correct"
65                       textScore.text = userscore.toString()
66                   }
67                   else
68                   {
69                       userlife--
70                       textQuestion.text = "Sorry,your answer is wrong"
71                       textLife.text = userlife.toString()
72                   }
73               }
74           }
75           buttonNext.setOnClickListener{
76               pauseTimer()
77               resetTImer()
78               gameContinue()
79               editTextAnswer.setText("")
80   
81               if(userlife == 0)
82               {
83                   Toast.makeText(applicationContext,"Game Over",Toast.LENGTH_LONG).show()
84                   val intent = Intent(this@GameActivity,ResultActivity::class.java)
85                   intent.putExtra("score",userscore)
86                   startActivity(intent)
87                   finish()
88               }
89               else
90               {
91                   gameContinue()
92               }
93   
94           }
95       }
96   
97       fun gameContinue()
98       {
99           val number1 = Random.nextInt(0,100)
100          val number2 = Random.nextInt(0,100)
101  
102          textQuestion.text = "$number1 + $number2"
103          correctanswer = number1 + number2
104  
105          startTimer()
106      }
107  
108      fun startTimer()
109      {
110          timer = object : CountDownTimer(timeLeftInmillis,1000){
111              override fun onTick(millisUntilFinished: Long) {
112                  timeLeftInmillis = millisUntilFinished
113                  updateText()
114              }
115  
116              override fun onFinish() {
117                  pauseTimer()
118                  resetTImer()
119                  updateText()
120  
121                  userlife--
122                  textLife.text = userlife.toString()
123                  textQuestion.text = "Sorry,your time is up!"
124              }
125  
126          }.start()
127  
128      }
129  
130      fun updateText()
131      {
132          val remainingTime : Int = (timeLeftInmillis/1000).toInt()
133          textTime.text = String.format(Locale.getDefault(),"%d",remainingTime)
134      }
135  
136      fun pauseTimer()
137      {
138              timer.cancel()
139      }
140  
141      fun resetTImer()
142      {
143          timeLeftInmillis = startTimerInmillis
144          updateText()
145      }
146  }
